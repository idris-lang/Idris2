0 | ||| Core features required to perform Golden file testing.
  1 | |||
  2 | ||| We provide the core functionality to run a *single* golden file test, or
  3 | ||| a whole test tree.
  4 | ||| This allows the developer freedom to use as is or design the rest of the
  5 | ||| test harness to their liking.
  6 | |||
  7 | ||| This was originally used as part of Idris2's own test suite and
  8 | ||| the core functionality is useful for the many and not the few.
  9 | ||| Please see Idris2 test harness for example usage.
 10 | |||
 11 | ||| # Test Structure
 12 | |||
 13 | ||| This harness works from the assumption that each individual golden test
 14 | ||| comprises of a directory with the following structure:
 15 | |||
 16 | ||| + `run` a *shell* script that runs the test. We expect it to:
 17 | |||   * Use `$1` as the variable standing for the idris executable to be tested
 18 | |||   * May use `${IDRIS2_TESTS_CG}` to pick a codegen that ought to work
 19 | |||   * Clean up after itself (e.g. by running `rm -rf build/`)
 20 | |||
 21 | ||| + `expected` a file containting the expected output of `run`
 22 | |||
 23 | ||| During testing, the test harness will generate an artefact named `output`
 24 | ||| and display both outputs if there is a failure.
 25 | ||| During an interactive session the following command is used to compare them
 26 | ||| as they are:
 27 | |||
 28 | ||| ```sh
 29 | |||  git diff --no-index --exit-code --word-diff-regex=. --color expected output
 30 | ||| ```
 31 | |||
 32 | ||| If `git` fails then the runner will simply present the expected and 'given'
 33 | ||| files side-by-side.
 34 | |||
 35 | ||| Of note, it is helpful to add `output` to a local `.gitignore` instance
 36 | ||| to ensure that it is not mistakenly versioned.
 37 | |||
 38 | ||| # Options
 39 | |||
 40 | ||| The test harness has several options that may be set:
 41 | |||
 42 | ||| + `idris2`       The path of the executable we are testing.
 43 | ||| + `codegen`      The backend to use for code generation.
 44 | ||| + `onlyNames`    The tests to run relative to the generated executable.
 45 | ||| + `onlyFile`     The file listing the tests to run relative to the generated executable.
 46 | ||| + `interactive`  Whether to offer to update the expected file or not.
 47 | ||| + `timing`       Whether to display time taken for each test.
 48 | ||| + `threads`      The maximum numbers to use (default: number of cores).
 49 | ||| + `failureFile`  The file in which to write the list of failing tests.
 50 | |||
 51 | ||| We provide an options parser (`options`) that takes the list of command line
 52 | ||| arguments and constructs this for you.
 53 | |||
 54 | ||| # Usage
 55 | |||
 56 | ||| When compiled to an executable the expected usage is:
 57 | |||
 58 | |||```sh
 59 | ||| runtests <path to executable under test>
 60 | |||   [--timing]
 61 | |||   [--interactive]
 62 | |||   [--only-file PATH]
 63 | |||   [--failure-file PATH]
 64 | |||   [--threads N]
 65 | |||   [--cg CODEGEN]
 66 | |||   [--only [NAMES]]
 67 | |||```
 68 | |||
 69 | ||| assuming that the test runner is compiled to an executable named `runtests`.

198 | ||| Run the specified Golden test with the supplied options.
199 | ||| See the module documentation for more information.
200 | ||| @testPath the directory that contains the test.
201 | export
202 | runTest : Options -> (testPath : String) -> IO Result
203 | runTest opts testPath = do
204 |   start <- clockTime UTC
205 |   let cg = maybe "" (" --cg " ++) (codegen opts)
206 |   let exe = "\"" ++ exeUnderTest opts ++ cg ++ "\""
207 |   ignore $ system $ "cd " ++ escapeArg testPath ++ " && " ++
208 |     "sh ./run " ++ exe ++ " | tr -d '\\r' > output"
209 |   end <- clockTime UTC
210 |
211 |   Right out <- readFile $ testPath ++ "/output"
212 |     | Left err => do putStrLn $ (testPath ++ "/output") ++ ": " ++ show err
213 |                      pure (Left testPath)
214 |
215 |   Right exp <- readFile $ testPath ++ "/expected"
216 |     | Left FileNotFound => do
217 |         if interactive opts
218 |           then mayOverwrite Nothing out
219 |           else putStrLn $ (testPath ++ "/expected") ++ ": " ++ show FileNotFound
220 |         pure (Left testPath)
221 |     | Left err => do putStrLn $ (testPath ++ "/expected") ++ ": " ++ show err
222 |                      pure (Left testPath)
223 |
224 |   let result = normalize out == normalize exp
225 |   let time = timeDifference end start
226 |
227 |   if result
228 |     then printTiming opts.timing time testPath $ maybeColored BrightGreen "success"
229 |     else do
230 |       printTiming opts.timing time testPath $ maybeColored BrightRed "FAILURE"
231 |       if interactive opts
232 |         then mayOverwrite (Just exp) out
233 |         else putStr . unlines $ expVsOut exp out
234 |
235 |   pure $ if result then Right testPath else Left testPath
236 |

263 |           , maybeColored BrightBlue out
264 |           , "Accept new golden value? [y/N]"
265 |           ]
266 |         Just exp => do
267 |           code <- system $ "git diff --no-index --exit-code --word-diff-regex=. " ++
268 |             (if opts.color then  "--color " else "") ++
269 |             escapeArg testPath ++ "/expected " ++ escapeArg testPath ++ "/output"
270 |           putStr . unlines $
271 |             ["Golden value differs from actual value."] ++
272 |             (if badSystemExitCode code then expVsOut exp out else []) ++
273 |             ["Accept actual value as new golden value? [y/N]"]
274 |       b <- getAnswer
275 |       when b $ do Right _ <- writeFile (testPath ++ "/expected") out
276 |                     | Left err => putStrLn $ (testPath ++ "/expected") ++ ": " ++ show err

279 |     printTiming : Bool -> Clock type -> String -> String -> IO ()
280 |     printTiming False _     path msg = putStrLn $ concat [path, ": ", msg]
281 |     printTiming True  clock path msg =
282 |       let time  = showTime 2 3 clock
283 |           -- We use 9 instead of `String.length msg` because:
284 |           -- 1. ": success" and ": FAILURE" have the same length
285 |           -- 2. ANSI escape codes make the msg look longer than it is
286 |           spent = String.length time + String.length path + 9
287 |           pad   = pack $ replicate (minus 72 spent) ' '
288 |       in putStrLn $ concat [path, ": ", msg, pad, time]

533 | ||| A runner for a test pool. If there are tests in the @TestPool@ that we want
534 | ||| to run, spawns `opts.threads` many runners and sends them the tests,
535 | ||| collecting all the results in the @Summary@ returned at the end.
536 | |||
537 | ||| @opts The options for the TestPool.
538 | ||| @pool The TestPool to run.
539 | export
540 | poolRunner : Options -> TestPool -> IO Summary
541 | poolRunner opts pool
542 |   = do -- check that we indeed want to run some of these tests
543 |        let tests = filterTests opts (testCases pool)
544 |        let (_ :: _) = tests
545 |              | [] => pure initSummary
546 |        -- if so make sure the constraints are satisfied
547 |        cs <- for (toList (codegen pool) ++ constraints pool) $ \ req => do
548 |           mfp <- checkRequirement req
549 |           let msg = case mfp of
550 |                       Nothing => "✗ " ++ show req ++ " not found"
551 |                       Just fp => "✓ Found " ++ show req ++ " at " ++ fp
552 |           pure (mfp, msg)
553 |        let (cs, msgs) = unzip cs
554 |
555 |        putStrLn (banner msgs)
556 |
557 |        let Just _ = the (Maybe (List String)) (sequence cs)
558 |              | Nothing => pure initSummary
559 |
560 |        -- if the test pool requires a specific codegen then use that
561 |        let opts = case codegen pool of
562 |                     Nothing => { codegen := Nothing } opts
563 |                     Just cg => { codegen := Just (show @{CG} cg) } opts
564 |                     Default => opts
565 |
566 |        -- set up the channels
567 |        accChan <- makeChannel
568 |        resChan <- makeChannel
569 |        testChan <- makeChannel
570 |
571 |        -- and then run all the tests
572 |
573 |        for_ (replicate opts.threads 0) $ \_ =>
574 |          fork (testThread opts testChan resChan)
575 |        -- start sending tests
576 |        senderTID <- fork $ testSender testChan opts.threads tests
577 |        -- start receiving results
578 |        receiverTID <- fork $ testReceiver resChan initSummary accChan opts.threads
579 |        -- wait until things are done, i.e. until we receive the final acc
580 |        acc <- channelGet accChan
581 |        pure acc
