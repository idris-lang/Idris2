 5 | ||| Run two linear computations in parallel and return the results.
 6 | export
 7 | par1 : L1 IO a -@ L1 IO b -@ L1 IO (LPair a b)
 8 | par1 x y
 9 |   = do aChan <- makeChannel
10 |        bChan <- makeChannel
11 |        aId <- liftIO1 $ fork $ withChannel aChan x
12 |        bId <- liftIO1 $ fork $ withChannel bChan y
13 |        a <- channelGet aChan
14 |        b <- channelGet bChan
15 |        pure1 (a # b)

18 |
19 |   -- This unsafe implementation temporarily bypasses the linearity checker.
20 |   -- However `par`'s implementation does not duplicate the values
21 |   -- and the type of `par` ensures that client code is not allowed to either!
22 |   withChannel : Channel t -> L1 IO t -@ IO ()
23 |   withChannel ch = assert_linear $ \ act => do
24 |     a <- LIO.run (act >>= assert_linear pure)
25 |     channelPut ch a
