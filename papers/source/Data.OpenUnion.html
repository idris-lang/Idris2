19 | ||| An open union of families is an index picking a family out together with
20 | ||| a value in the family thus picked.
21 | public export
22 | data Union : (elt : a -> Type) -> (ts : List a) -> Type where
23 |   Element : (k : Nat) -> (0 _ : AtIndex t ts k) -> elt t -> Union elt ts
24 |
25 | ||| An empty open union of families is empty
26 | public export
27 | Uninhabited (Union elt []) where
28 |   uninhabited (Element _ p _) = void (uninhabited p)

30 | ||| Injecting a value into an open union, provided we know the index of
31 | ||| the appropriate type family.
32 | inj' : (k : Nat) -> (0 _ : AtIndex t ts k) -> elt t -> Union elt ts
33 | inj' = Element

35 | ||| Projecting out of an open union, provided we know the index of the
36 | ||| appropriate type family. This may obviously fail if the value stored
37 | ||| actually corresponds to another family.
38 | prj' : (k : Nat) -> (0 _ : AtIndex t ts k) -> Union elt ts -> Maybe (elt t)
39 | prj' k p (Element k' q t) with (decEq k  k')
40 |   prj' k p (Element k q t) | Yes Refl = rewrite atIndexUnique p q in Just t
41 |   prj' k p (Element k' q t) | No neq = Nothing

43 | ||| Given that equality of type families is not decidable, we have to
44 | ||| rely on the interface `Member` to automatically find the index of a
45 | ||| given family.
46 | public export
47 | inj : Member t ts => elt t -> Union elt ts
48 | inj = let (Element n p) = isMember t ts in inj' n p

50 | ||| Given that equality of type families is not decidable, we have to

57 | ||| By doing a bit of arithmetic we can figure out whether the union's value
58 | ||| came from the left or the right list used in the index.
59 | public export
60 | split : Subset Nat (flip HasLength ss) ->
61 |         Union elt (ss ++ ts) -> Either (Union elt ss) (Union elt ts)
62 | split m (Element n p t) with (@@ lt n (fst m))
63 |   split m (Element n p t) | (True ** lt)
64 |     = Left (Element n (strengthenL m lt p) t)
65 |   split m (Element n p t) | (False ** notlt)
66 |     = let 0 lte : lte (fst m) n === True
67 |                 = LteIslte (fst m) n (notltIsGTE n (fst m) notlt)
68 |       in Right (Element (minus n (fst m)) (strengthenR m lte p) t)

70 | ||| We can inspect an open union over a non-empty list of families to check
71 | ||| whether the value it contains belongs either to the first family or any
72 | ||| other in the tail.
73 | public export
74 | decomp : Union elt (t :: ts) -> Either (Union elt ts) (elt t)
75 | decomp (Element 0     (Z)   t) = Right t
76 | decomp (Element (S n) (S p) t) = Left (Element n p t)
77 |
78 | ||| An open union over a singleton list is just a wrapper
79 | public export
80 | decomp0 : Union elt [t] -> elt t
81 | decomp0 elt = case decomp elt of
82 |   Left t => absurd t
83 |   Right t => t

85 | ||| Inserting new union members on the right leaves the index unchanged.
86 | public export
87 | weakenR : Union elt ts -> Union elt (ts ++ us)
88 | weakenR (Element n p t) = Element n (weakenR p) t
89 |
90 | ||| Inserting new union members on the left, requires shifting the index by
91 | ||| the number of members introduced. Note that this number is the only
92 | ||| thing we need to keep around at runtime.
93 | public export
94 | weakenL : Subset Nat (flip HasLength ss) -> Union elt ts -> Union elt (ss ++ ts)
95 | weakenL m (Element n p t) = Element (fst m + n) (weakenL m p) t
