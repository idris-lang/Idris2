 24 | ||| Equal non-empty lists should result in equal components after destructuring 'snoc'.
 25 | export
 26 | snocInjective : {x : a} -> {xs : List a} -> {y : a} -> {ys : List a} ->
 27 |             (xs `snoc` x) = (ys `snoc` y) -> (xs = ys, x = y)
 28 | snocInjective {xs = []} {ys = []} Refl = (Refl, Refl)
 29 | snocInjective {xs = []} {ys = z :: zs} prf =
 30 |   let nilIsSnoc = snd $ consInjective {xs = []} {ys = zs ++ [y]} prf
 31 |    in void $ snocNonEmpty (sym nilIsSnoc)
 32 | snocInjective {xs = z :: xs} {ys = []} prf =
 33 |   let snocIsNil = snd $ consInjective {x = z} {xs = xs ++ [x]} {ys = []} prf
 34 |    in void $ snocNonEmpty snocIsNil
 35 | snocInjective {xs = w :: xs} {ys = z :: ys} prf =
 36 |   let (wEqualsZ, xsSnocXEqualsYsSnocY) = consInjective prf
 37 |       (xsEqualsYS, xEqualsY) = snocInjective xsSnocXEqualsYsSnocY
 38 |    in (consCong2 wEqualsZ xsEqualsYS, xEqualsY)
