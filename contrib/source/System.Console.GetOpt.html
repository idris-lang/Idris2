0 | ||| This library provides facilities for parsing the command-line options
  1 | ||| in a standalone program. It is essentially an Idris port of the GNU getopt library.
  2 | ||| (Actually, it is an Idris port of the [corresponding Haskell module]
  3 | ||| (http://hackage.haskell.org/package/base-4.14.1.0/docs/System-Console-GetOpt.html)).
  4 | module System.Console.GetOpt
  5 |
  6 | import Data.List

 12 | ||| What to do with options following non-options
 13 | public export
 14 | data ArgOrder a =
 15 |   ||| no option processing after first non-option
 16 |   RequireOrder                |
 17 |   ||| freely intersperse options and non-options
 18 |   Permute                     |
 19 |   ||| wrap non-options into options
 20 |   ReturnInOrder (String -> a)
 21 |
 22 | export
 23 | Functor ArgOrder where
 24 |   map _ RequireOrder      = RequireOrder
 25 |   map _ Permute           = Permute
 26 |   map f (ReturnInOrder g) = ReturnInOrder (f . g)
 27 |
 28 | ||| Describes whether an option takes an argument or not, and if so
 29 | ||| how the argument is injected into a value of type `a`.
 30 | public export
 31 | data ArgDescr a =
 32 |    ||| no argument expected
 33 |    NoArg                   a         |
 34 |    ||| option requires argument
 35 |    ReqArg (String       -> a) String |
 36 |    ||| optional argument
 37 |    OptArg (Maybe String -> a) String
 38 |
 39 | export
 40 | Functor ArgDescr where
 41 |   map f (NoArg x)    = NoArg (f x)
 42 |   map f (ReqArg g x) = ReqArg (f . g) x
 43 |   map f (OptArg g x) = OptArg (f . g) x
 44 |
 45 | ||| Each `OptDescr` describes a single option.
 46 | |||
 47 | ||| The arguments to 'Option' are:
 48 | |||
 49 | ||| * list of short option characters
 50 | ||| * list of long option strings (without \"--\")
 51 | ||| * argument descriptor
 52 | ||| * explanation of option for user
 53 | public export
 54 | record OptDescr a where
 55 |   constructor MkOpt
 56 |   ||| list of short option characters
 57 |   shortNames  : List Char
 58 |   ||| list of long option strings (without "--")
 59 |   longNames   : List String
 60 |   ||| argument descriptor
 61 |   argDescr    : ArgDescr a
 62 |   ||| explanation of option for user
 63 |   description : String
 64 |
 65 | export
 66 | Functor OptDescr where
 67 |   map f = { argDescr $= map f }
 68 |
 69 | -- kind of cmd line arg (internal use only):
 70 | data OptKind a
 71 |    = Opt       a                --    an option
 72 |    | UnreqOpt  String           --    an un-recognized option
 73 |    | NonOpt    String           --    a non-option
 74 |    | EndOfOpts                  --    end-of-options marker (i.e. "--")
 75 |    | OptErr    String           --    something went wrong...
 76 |
 77 | --------------------------------------------------------------------------------
 78 | --          Printing Usage Info
 79 | --------------------------------------------------------------------------------
 80 |
 81 | fmtShort : ArgDescr a -> Char -> String
 82 | fmtShort (NoArg  _   ) so = "-" ++ singleton so
 83 | fmtShort (ReqArg _ ad) so = "-" ++ singleton so ++ " " ++ ad
 84 | fmtShort (OptArg _ ad) so = "-" ++ singleton so ++ "[" ++ ad ++ "]"
 85 |
 86 | fmtLong : ArgDescr a -> String -> String
 87 | fmtLong (NoArg  _   ) lo = "--" ++ lo
 88 | fmtLong (ReqArg _ ad) lo = "--" ++ lo ++ "=" ++ ad
 89 | fmtLong (OptArg _ ad) lo = "--" ++ lo ++ "[=" ++ ad ++ "]"
 90 |
 91 | fmtOpt : OptDescr a -> List (String,String,String)
 92 | fmtOpt (MkOpt sos los ad descr) =
 93 |   let sosFmt = concat $ intersperse ", " (map (fmtShort ad) sos)
 94 |       losFmt = concat $ intersperse ", " (map (fmtLong ad) los)
 95 |       (h ::: t) = lines1 descr in
 96 |       (sosFmt,losFmt,h) :: map (\s => ("","",s)) t
 97 |   where lines1 : String -> List1 String
 98 |         lines1 s = case lines s of
 99 |                         [] => "" ::: []
100 |                         (x :: xs) => x ::: xs

102 | ||| Return a string describing the usage of a command, derived from
103 | ||| the header (first argument) and the options described by the
104 | ||| second argument.
105 | public export
106 | usageInfo : (header : String) -> List (OptDescr a) -> String
107 | usageInfo header optDescr =
108 |   let (ss,ls,ds)   = (unzip3 . concatMap fmtOpt) optDescr
109 |       paste        = \x,y,z => "  " ++ x ++ "  " ++ y ++ "  " ++ z
110 |       table        = zipWith3 paste (sameLen ss) (sameLen ls) ds
111 |    in unlines $ header :: table
112 |
113 |   where flushLeft : Nat -> String -> String
114 |         flushLeft n s = s ++ pack (replicate (n `minus` length s) ' ')
115 |
116 |         sameLen : List String -> List String
117 |         sameLen ss = let len = foldl (\n => max n . length) 0 ss
118 |                       in map (flushLeft len) ss
119 |

125 | errAmbig ods s = let h = "option `" ++ s ++ "' is ambiguous; could be one of:"
126 |                   in OptErr (usageInfo h ods)

128 | errReq : String -> (optStr : String) -> OptKind a

130 |

138 | ||| Result of parsing the command line arguments accoring to a list
139 | ||| of `OptDescr`s. (see also function `getOpt`).
140 | public export
141 | record Result a where
142 |   constructor MkResult
143 |   ||| List of successfully parsed options
144 |   options      : List a
145 |
146 |   ||| List of non-options (other command line arguments)
147 |   nonOptions   : List String
148 |
149 |   ||| List of unrecognized options.
150 |   unrecognized : List String
151 |
152 |   ||| Errors during option parsing. These occur, for instance, when
153 |   ||| an option requires an additional argument but none was given.
154 |   errors       : List String
155 |
156 | public export
157 | emptyRes : Result a
158 | emptyRes = MkResult [] [] [] []
159 |
160 | export
161 | Functor Result where
162 |   map f = { options $= map f }
163 |
164 | OptFun : Type -> Type
165 | OptFun a = List String -> List (OptDescr a) -> (OptKind a,List String)
166 |
167 | longOpt : String -> OptFun a
168 | longOpt ls rs descs =
169 |   let (opt,arg) = break ('=' ==) ls
170 |       getWith   = \p => filter (any (p opt) . longNames) descs
171 |       exact     = getWith (==)
172 |       options   = if null exact then getWith isPrefixOf else exact
173 |       ads       = map argDescr options
174 |       os        = "--" ++ opt
175 |    in case (ads,unpack arg,rs) of
176 |            (_ :: _ :: _ , _      ,  r        ) => (errAmbig options os, r)
177 |            ([NoArg  a  ], []     ,  r        ) => (Opt a, r)
178 |            ([NoArg  a  ], c :: _ ,  r        ) => (errNoArg os,r)
179 | --                        ^ this is known (but not proven) to be '='
180 |
181 |            ([ReqArg _ d], []     ,  []       ) => (errReq d os,[])
182 |            ([ReqArg f _], []     ,  (r::rest)) => (Opt $ f r,rest)
183 |            ([ReqArg f _], c :: xs,  r        ) => (Opt $ f (pack xs),r)
184 | --                        ^ this is known (but not proven) to be '='
185 |
186 |            ([OptArg f _], []     ,  r        ) => (Opt $ f Nothing,r)
187 |            ([OptArg f _], c :: xs,  r        ) => (Opt . f . Just $ pack xs,r)
188 | --                        ^ this is known (but not proven) to be '='
189 |
190 |            ([]          , _      ,  r        ) => (UnreqOpt $ "--" ++ ls,r)
191 |
192 | shortOpt : Char -> String -> OptFun a
193 | shortOpt y ys rs descs =
194 |   let options = filter (elem y . shortNames) descs
195 |       ads     = map argDescr options
196 |       mkOs    = strCons '-'
197 |       os      = mkOs (singleton y)
198 |    in case (ads,ys,rs) of
199 |            (_ :: _ :: _ , _ ,  r        ) => (errAmbig options os, r)
200 |            ([NoArg  a  ], "",  r        ) => (Opt a,r)
201 |            ([NoArg  a  ], s ,  r        ) => (Opt a, mkOs s :: r)
202 |            ([ReqArg _ d], "",  []       ) => (errReq d os, [])
203 |            ([ReqArg f _], "",  (r::rest)) => (Opt $ f r, rest)
204 |            ([ReqArg f _], s ,  r        ) => (Opt $ f s, r)
205 |            ([OptArg f _], "",  r        ) => (Opt $ f Nothing, r)
206 |            ([OptArg f _], s ,  r        ) => (Opt . f $ Just s, r)
207 |            ([]          , "",  r        ) => (UnreqOpt os, r)
208 |            ([]          , s ,  r        ) => (UnreqOpt os, mkOs s :: r)
209 |
210 |
211 | -- take a look at the next cmd line arg and decide what to do with it
212 | getNext : List Char -> OptFun a
213 | getNext ('-'::'-'::[]) r _        = (EndOfOpts,r)
214 | getNext ('-'::'-'::xs) r descs    = longOpt (pack xs) r descs
215 | getNext ('-':: x ::xs) r descs    = shortOpt x (pack xs) r descs
216 | getNext a              r _        = (NonOpt $ pack a,r)

218 | ||| Process the command-line, and return the list of values that matched
219 | ||| (and those that didn't). The arguments are:
220 | |||
221 | ||| * The order requirements (see `ArgOrder`)
222 | |||
223 | ||| * The option descriptions (see `OptDescr`)
224 | |||
225 | ||| * The actual command line arguments (presumably got from
226 | |||   `System.getArgs`).
227 | export
228 | getOpt :  ArgOrder a                         -- non-option handling
229 |        -> List (OptDescr a)                  -- option descriptors
230 |        -> (args : List String)               -- the command-line arguments
231 |        -> Result a
232 | getOpt _        _        []         =  emptyRes
233 | getOpt ordering descs (arg::args)   =
234 |   let (opt,rest) = getNext (unpack arg) args descs
235 |       res        = getOpt ordering descs (assert_smaller args rest)
236 |    in case (opt,ordering) of
237 |            (Opt x, _)                   => {options $= (x::)} res
238 |            (UnreqOpt x, _)              => {unrecognized $= (x::)} res
239 |            (NonOpt x, RequireOrder)     => MkResult [] (x::rest) [] []
240 |            (NonOpt x, Permute)          => {nonOptions $= (x::)} res
241 |            (NonOpt x, ReturnInOrder f)  => {options $= (f x::)} res
242 |            (EndOfOpts, RequireOrder)    => MkResult [] rest [] []
243 |            (EndOfOpts, Permute)         => MkResult [] rest [] []
244 |            (EndOfOpts, ReturnInOrder f) => MkResult (map f rest) [] [] []
245 |            (OptErr e, _)                => {errors $= (e::)} res
