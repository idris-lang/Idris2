  8 | ||| `Validated` is like an `Either` but accumulates all errors with semigroup operation.
  9 | public export
 10 | data Validated e a = Valid a | Invalid e

 14 | public export
 15 | (Eq e, Eq a) => Eq (Validated e a) where
 16 |   Valid x   == Valid y   = x == y
 17 |   Invalid e == Invalid f = e == f

 50 | ||| Applicative composition preserves invalidity sequentially accumulating all errors.
 51 | public export
 52 | Semigroup e => Applicative (Validated e) where
 53 |   pure = Valid
 54 |
 55 |   Valid f    <*> Valid x    = Valid $ f x
 56 |   Invalid e1 <*> Invalid e2 = Invalid $ e1 <+> e2
 57 |   Invalid e  <*> Valid _    = Invalid e
 58 |   Valid _    <*> Invalid e  = Invalid e
 59 |
 60 | -- There is no `Monad` implementation because it can't be coherent with the accumulating `Applicative` one.
 61 |
 62 | ||| Semigroup operation selects the leftmost valid value.
 63 | ||| If both sides are invalid, errors are accumulated.
 64 | public export
 65 | Semigroup e => Semigroup (Validated e a) where
 66 |   l@(Valid _) <+> _           = l
 67 |   _           <+> r@(Valid _) = r
 68 |   Invalid e1  <+> Invalid e2  = Invalid $ e1 <+> e2
 69 |
 70 | public export
 71 | Monoid e => Monoid (Validated e a) where
 72 |   neutral = Invalid neutral
 73 |
 74 | ||| Alternative composition preserves validity selecting the leftmost valid value.
 75 | ||| If both sides are invalid, errors are accumulated.
 76 | public export
 77 | Monoid e => Alternative (Validated e) where
 78 |   empty = neutral
 79 |   l@(Valid _) <|> _           = l
 80 |   _           <|> r@(Valid _) = r
 81 |   Invalid e1  <|> Invalid e2  = Invalid $ e1 <+> e2
 82 |
 83 | public export
 84 | Foldable (Validated e) where
 85 |   foldr op init $ Valid x   = x `op` init
 86 |   foldr _  init $ Invalid _ = init
 87 |
 88 |   foldl op init $ Valid x   = init `op` x
 89 |   foldl _  init $ Invalid _ = init
 90 |
 91 |   null $ Valid _   = False
 92 |   null $ Invalid _ = True
 93 |
 94 | public export
 95 | Traversable (Validated e) where
 96 |   traverse f $ Valid x   = Valid <$> f x
 97 |   traverse _ $ Invalid e = pure $ Invalid e
 98 |
 99 | public export
100 | Semigroup e => Zippable (Validated e) where
101 |   zipWith f (Valid l)   (Valid r)   = Valid $ f l r
102 |   zipWith _ (Valid _)   (Invalid r) = Invalid r
103 |   zipWith _ (Invalid l) (Valid _)   = Invalid l
104 |   zipWith _ (Invalid l) (Invalid r) = Invalid $ l <+> r
105 |
106 |   zipWith3 f (Valid x)   (Valid y)   (Valid z)   = Valid $ f x y z
107 |   zipWith3 _ (Valid _)   (Valid _)   (Invalid z) = Invalid z
108 |   zipWith3 _ (Valid _)   (Invalid y) (Valid _)   = Invalid y
109 |   zipWith3 _ (Valid _)   (Invalid y) (Invalid z) = Invalid $ y <+> z
110 |   zipWith3 _ (Invalid x) (Valid _)   (Valid _)   = Invalid x
111 |   zipWith3 _ (Invalid x) (Valid _)   (Invalid z) = Invalid $ x <+> z
112 |   zipWith3 _ (Invalid x) (Invalid y) (Valid _)   = Invalid $ x <+> y
113 |   zipWith3 _ (Invalid x) (Invalid y) (Invalid z) = Invalid $ x <+> y <+> z
114 |
115 |   unzipWith f (Valid x)   = let (a, b) = f x in (Valid a, Valid b)
116 |   unzipWith _ (Invalid e) = (Invalid e, Invalid e)
117 |
118 |   unzipWith3 f (Valid x)   = let (a, b, c) = f x in (Valid a, Valid b, Valid c)
119 |   unzipWith3 _ (Invalid e) = (Invalid e, Invalid e, Invalid e)
