#+begin_src idris
data Vect : Nat -> Type -> Type where
     Nil : Vect Z a
     (::) : a -> Vect k a -> Vect (S k) a
#+end_src

#+IDRIS: %name Vect xs, ys, zs

#+begin_src idris
append : Vect n a -> Vect m a -> Vect (n + m) a
append {n} xs ys = ?foo
#+end_src

#+begin_src idris
vadd : Num a => Vect n a -> Vect n a -> Vect n a
vadd [] ys = ?bar
vadd (x :: xs) ys = ?baz
#+end_src

#+begin_src idris
suc : (x : Nat) -> (y : Nat) -> x = y -> S x = S y
suc x y prf = ?quux
#+end_src

#+begin_src idris
suc' : x = y -> S x = S y
suc' {x} {y} prf = ?quuz
#+end_src
