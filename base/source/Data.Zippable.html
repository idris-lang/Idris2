 8 | ||| The `Zippable` interface describes how you can combine and split the
 9 | ||| elements in a parameterised type.
10 | ||| @ z the parameterised type
11 | public export
12 | interface Zippable z where
13 |   ||| Combine two parameterised types by applying a function.
14 |   ||| @ z the parameterised type
15 |   ||| @ func the function to combine elements with
16 |   zipWith : (func : a -> b -> c) -> z a -> z b -> z c
17 |
18 |   ||| Combine two parameterised types into a parameterised type of pairs.
19 |   ||| @ z the parameterised type
20 |   zip : z a -> z b -> z (a, b)
21 |   zip = zipWith (,)
22 |
23 |   export infixr 6 `zip`
24 |
25 |   ||| Combine three parameterised types by applying a function.
26 |   ||| @ z the parameterised type
27 |   ||| @ func the function to combine elements with
28 |   zipWith3 : (func : a -> b -> c -> d) -> z a -> z b -> z c -> z d
29 |
30 |   ||| Combine three parameterised types into a parameterised type of triplets.
31 |   ||| @ z the parameterised type
32 |   zip3 : z a -> z b -> z c -> z (a, b, c)
33 |   zip3 = zipWith3 (,,)
34 |
35 |   ||| Split a parameterised type by applying a function into a pair of
36 |   ||| parameterised types.
37 |   ||| @ z the parameterised type
38 |   ||| @ func the function to split elements with
39 |   unzipWith : (func : a -> (b, c)) -> z a -> (z b, z c)
40 |
41 |   ||| Split a parameterised type of pairs into a pair of parameterised types.
42 |   ||| @ z the parameterised type
43 |   unzip : z (a, b) -> (z a, z b)
44 |   unzip = unzipWith id
45 |
46 |   ||| Split a parameterised type by applying a function into a triplet of
47 |   ||| parameterised types.
48 |   ||| @ z the parameterised type
49 |   ||| @ func the function to split elements with
50 |   unzipWith3 : (func : a -> (b, c, d)) -> z a -> (z b, z c, z d)
51 |
52 |   ||| Split a parameterised type of triplets into a triplet of parameterised
53 |   ||| types.
54 |   ||| @ z the parameterised type
55 |   unzip3 : z (a, b, c) -> (z a, z b, z c)
56 |   unzip3 = unzipWith3 id

58 | public export
59 | [Compose] Zippable f => Zippable g => Zippable (f . g) where
60 |   zipWith f = zipWith $ zipWith f
61 |   zipWith3 f = zipWith3 $ zipWith3 f
62 |   unzipWith f = unzipWith $ unzipWith f
63 |   unzipWith3 f = unzipWith3 $ unzipWith3 f
64 |
65 |   zip = zipWith zip
66 |   zip3 = zipWith3 zip3
67 |   unzip = unzipWith unzip
68 |   unzip3 = unzipWith3 unzip3
69 |
70 | ||| Variant of composing and decomposing using existing `Applicative` implementation.
71 | |||
72 | ||| This implementation is lazy during unzipping.
73 | ||| Caution! It may repeat the whole original work, each time the unzipped elements are used.
74 | |||
75 | ||| Please be aware that not every `Applicative` has the same semantics as `Zippable`.
76 | ||| Consider `List` as a simple example.
77 | ||| However, this implementation is applicable for lazy data types or deferred computations.
78 | public export
79 | [FromApplicative] Applicative f => Zippable f where
80 |   zipWith f x y = [| f x y |]
81 |   zipWith3 f x y z = [| f x y z |]
82 |
83 |   unzip u = (map fst u, map snd u)
84 |   unzip3 u = (map fst u, map (fst . snd) u, map (snd . snd) u)
85 |   unzipWith f = unzip . map f
86 |   unzipWith3 f = unzip3 . map f
87 |
88 |  -- To be moved appropriately as soon as we have some module like `Data.Pair`, like other prelude types have.
89 | public export
90 | Semigroup a => Zippable (Pair a) where
91 |   zipWith f (l, x) (r, y) = (l <+> r, f x y)
92 |   zipWith3 f (l, x) (m, y) (r, z) = (l <+> m <+> r, f x y z)
93 |
94 |   unzipWith f (l, x) = bimap (l,) (l,) (f x)
95 |   unzipWith3 f (l, x) = bimap (l,) (bimap (l,) (l,)) (f x)
